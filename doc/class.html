Class

import { makeIterable, each } from './util';

addClass

$('.item').addClass('bar');

Parameters:

  • value must be a String.
    (The class name to add to the element(s).)

Returns a $Object
(or Node/List in native mode)

function addClass(value) { each(this, function(element) { element.classList.add(value); }); return this; }

removeClass

$('.items').removeClass('bar');

Parameters:

  • value must be a String.
    (The class name to remove from the element(s).)

Returns a $Object
(or Node/List in native mode)

function removeClass(value) { each(this, function(element) { element.classList.remove(value); }); return this; }

toggleClass

$('.item').toggleClass('bar');

Parameters:

  • value must be a String.
    (The class name to toggle at the element(s).)

Returns a $Object
(or Node/List in native mode)

function toggleClass(value) { each(this, function(element) { element.classList.toggle(value); }); return this; }

hasClass

$('.item').hasClass('bar');

Parameters:

  • value must be a String.
    (Check if the DOM element contains the class name. When applied to multiple elements, returns true if any of them contains the class name.)

Returns a boolean

function hasClass(value) { return makeIterable(this).some(function(element) { return element.classList.contains(value); }); }

Export interface

export { addClass, removeClass, toggleClass, hasClass };
